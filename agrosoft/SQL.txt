
CREATE TABLE CATEGORIES
(
	ID INT IDENTITY(1,1) CONSTRAINT HEADERS_ID_NN NOT NULL CONSTRAINT CATEGORIES_ID_PK PRIMARY KEY(ID),
	NAME NVARCHAR(100) CONSTRAINT CATEGORIES_NAME_N NULL ,
	ACTIVE BIT CONSTRAINT HEADERS_ACTIVE_N NULL
);

CREATE TABLE CUSTOMERS
(
	ID INT IDENTITY(1,1) CONSTRAINT CUSTOMERS_ID_NN NOT NULL CONSTRAINT CUSTOMERS_ID_PK PRIMARY KEY(ID),
	NAME NVARCHAR(100) CONSTRAINT CUSTOMERS_NAME_N NULL ,
	SURNAME NVARCHAR(100) CONSTRAINT CUSTOMERS_SURNAME_N NULL ,
	PHONE INT CONSTRAINT CUSTOMERS_PHONE_N NULL ,
	ADDRESS NVARCHAR(100) CONSTRAINT CUSTOMERS_ADDRESS_N NULL ,
	ACTIVE BIT CONSTRAINT HEADERS_ACTIVE_N NULL
);

CREATE TABLE ROLES
(
	ID INT IDENTITY(1,1) CONSTRAINT ROLES_ID_NN NOT NULL CONSTRAINT ROLES_ID_PK PRIMARY KEY(ID),
	ROLESNAME NVARCHAR(100) CONSTRAINT ROLES_NAME_N NULL
);

CREATE TABLE USERS
(
	ID INT IDENTITY(1,1) CONSTRAINT USERS_ID_NN NOT NULL CONSTRAINT USERS_ID_PK PRIMARY KEY(ID),
	ROLESID INT CONSTRAINT USERS_ROLESID_NN NOT NULL CONSTRAINT USERS_ROLESID_FK REFERENCES ROLES(ID),
	NAME NVARCHAR(100) CONSTRAINT USERS_NAME_N NULL ,
	SURNAME NVARCHAR(100) CONSTRAINT USERS_SURNAME_N NULL ,
	LOGIN NVARCHAR(100) CONSTRAINT USERS_LOGIN_N NULL ,
	PASSWORD NVARCHAR(100) CONSTRAINT USERS_PASSWORD_N NULL

);


CREATE TABLE PRODUCTS
(
	ID INT IDENTITY(1,1) CONSTRAINT PRODUCTS_ID_NN NOT NULL CONSTRAINT PRODUCTS_ID_PK PRIMARY KEY(ID),
	CATEGORIESID INT CONSTRAINT PRODUCTS_CATEGORIESID_NN NOT NULL CONSTRAINT PRODUCTS_CATEGORIESID_FK REFERENCES CATEGORIES(ID),
	CUSTOMERSID INT CONSTRAINT PRODUCTS_CUSTOMERSID_NN NOT NULL CONSTRAINT PRODUCTS_CUSTOMERSID_FK REFERENCES CUSTOMERS(ID),
	MODELNAME NVARCHAR(100) CONSTRAINT PRODUCTS_MODELNAME_N NULL,
	WARANTY INT CONSTRAINT PRODUCTS_WARANTY_N NULL,
	DATESOLD DATE CONSTRAINT PRODUCTS_DATESOLD_N NULL,
	SERIAL INT CONSTRAINT PRODUCTS_SERIAL_N NULL,
	TIMESPENT INT CONSTRAINT PRODUCTS_TIMESPENT_N NULL,
	PRICE INT CONSTRAINT PRODUCTS_PRICE_N NULL,
	SELFPRICE INT CONSTRAINT PRODUCTS_SELFPRICE_N NULL,
	SELFPRICECLIENT INT CONSTRAINT PRODUCTS_SELFPRICECLIENT_N NULL,
	HEIGHT INT CONSTRAINT PRODUCTS_HEIGHT_N NULL,
	WEIGHT INT CONSTRAINT PRODUCTS_WEIGHT_N NULL,
	VOLTAGE INT CONSTRAINT PRODUCTS_VOLTAGE_N NULL,
	RPM INT CONSTRAINT PRODUCTS_RPM_N NULL,
	PERFORMANCE INT CONSTRAINT PRODUCTS_PERFORMANCE_N NULL,
	ENGINEPOWER INT CONSTRAINT PRODUCTS_ENGINEPOWER_N NULL,
	HUMMERS	INT CONSTRAINT PRODUCTS_HUMMERS_N NULL,
	COMMENT NVARCHAR(MAX) CONSTRAINT PRODUCTS_COMMENT_N NULL,
	ACTIVE BIT CONSTRAINT PRODUCTS_ACTIVE_NN NOT NULL
);
