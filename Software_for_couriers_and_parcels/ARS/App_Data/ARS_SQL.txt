CREATE TABLE ORDERS
(
	ID INT IDENTITY(1,1) CONSTRAINT ORDERS_ID_NN NOT NULL CONSTRAINT ORDERS_ID_PK PRIMARY KEY(ID),
	DATE DATE CONSTRAINT ORDERS_DATE_N NULL,
	TIME NVARCHAR(10) CONSTRAINT ORDERS_TIME_N NULL,
	COMPANY	NVARCHAR(100) CONSTRAINT ORDERS_COMPANY_N NULL,
	CN INT CONSTRAINT ORDERS_CN_N NULL,
	PCS INT CONSTRAINT ORDERS_PCS_N NULL,
	ADDRESS NVARCHAR(100) CONSTRAINT ORDERS_ADDRESS_N NULL,
	PH INT CONSTRAINT ORDERS_PH_N NULL,
	CLIENTNAME NVARCHAR(100) CONSTRAINT ORDERS_CLIENTNAME_N NULL,	
	LOCAL NVARCHAR(100) CONSTRAINT ORDERS_LOCAL_N NULL,
	INTERN NVARCHAR(100) CONSTRAINT ORDERS_INTERN_N NULL,
	TYPE NVARCHAR(100) CONSTRAINT ORDERS_TYPE_N NULL,
	DESTINATION NVARCHAR(MAX) CONSTRAINT ORDERS_DESTINATION_N NULL
);
